Introducción

La desagregación hacia causas específicas (código a 3–4 dígitos CIE-10) permite identificar patrones imposibles de detectar a nivel de grandes grupos. Se analizan, entre otras : cardiopatía isquémica, cáncer de mama, diabetes tipo 2, cirrosis hepática, homicidios y accidentes de tránsito.

## Introducción

La desagregación hacia **causas específicas** (código a 3–4 dígitos CIE-10) permite
identificar patrones imposibles de detectar a nivel de grandes grupos.
Se analizan, entre otras : cardiopatía isquémica, cáncer de mama, diabetes tipo 2,
cirrosis hepática, homicidios y accidentes de tránsito.

annees <- c(1990, 1995, 2000, 2005, 2010, 2015, 2019, 2024)

# Étape 3 : Pivot
wide_hommes <- data_hommes |>
  select(CAP_GRUPO, year, asmr_100k) |>
  pivot_wider(names_from = year, values_from = asmr_100k) |>
  arrange(desc(`2024`))

wide_hommes <- wide_hommes |>
  left_join(dict_cause, by = "CAP_GRUPO") |>
  select(CAP_GRUPO, DESCRIP, everything())

# wide_hommes

# Étape 4 : Tableau
library(knitr)
library(kableExtra)
format_table <- function(wide_data, caption_txt) {
  wide_data |>
    mutate(across(where(is.numeric), ~ format(round(.x, 1), big.mark = ","))) |>
    kable(
      caption   = caption_txt,
      col.names = c("Cause", "Description", names(wide_data)[-(1:2)]),
      align     = c("l", "l", rep("r", ncol(wide_data) - 2))
    ) |>
    kable_styling(
      bootstrap_options = c("striped", "hover", "condensed"),
      full_width        = FALSE,
      font_size         = 13
    ) |>
    add_header_above(
      c(" " = 2,
        "Taux de mortalité standardisé (pour 100 000 hab.)" = ncol(wide_data) - 2)
    ) |>
    row_spec(1, bold = TRUE, background = "#f0f4f8")
}
tbl_hommes <- format_table(wide_hommes, "Top 10 causes de décès — Hommes")

tbl_hommes
